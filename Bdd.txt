CREATE DATABASE zahavamada;

USE zahavamada;

CREATE TABLE mpampiasa (
    id INT AUTO_INCREMENT PRIMARY KEY,
    anarana VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    tenimiafina VARCHAR(255) NOT NULL,
    daty_namoronana TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE toerana (
    id INT AUTO_INCREMENT PRIMARY KEY,
    anarana VARCHAR(100) NOT NULL,
    lat DOUBLE NOT NULL,
    lon DOUBLE NOT NULL,
    famaritana TEXT
);
